                    var objectPoolFactory = function(createFn) {
  var objectPool = []

  return {
    create: function() {
      var obj =
        objectPool.length === 0
          ? createFn.apply(this, arguments)
          : objectPool.shift()

      return obj
    },
    recover: function(obj) {
      objectPool.push(obj)
    },
  }
}
